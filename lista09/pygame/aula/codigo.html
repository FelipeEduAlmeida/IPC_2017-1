23   # função que toca arquivo de som 
24   def play(caminho_arquivo): 
25       pygame.mixer.music.load(caminho_arquivo) 
26       pygame.mixer.music.set_volume(0.2) 
27       pygame.mixer.music.play()

31   def mensagem(texto): 
32       textsurface = fonte30.render(texto, 1, WHITE) 
33       tela.blit(textsurface, (10, 600))

42   # função que finaliza o jogo 
43   def finaliza(texto): 
44       fundo = pygame.image.load("img/fundo.png").convert_alpha() 
45       tela.blit(fundo, pygame.rect.Rect(0, 0, 128, 128)) 
46       pygame.display.flip() 
47       text = fonte50.render(texto, 1, WHITE) 
48       text_rect = text.get_rect() 
49       tela.blit(text, [100, 250])

52   # funcao que verifica se algum jogador ganhou 
53   def verifica_se_ganhou(jogada): 
54       vitorias_possiveis = [[0, 1, 2], [3, 4, 5], [6, 7, 8], [0, 3, 6], [1, 4, 7], [2, 5, 8], [0, 4, 8], [2, 4, 6]] 
55    
56       for j in vitorias_possiveis: 
57           vezes = 0 
58           for l in j: 
59               for i in jogada: 
60    
61                   if l == i: 
62                       vezes += 1 
63    
64           if vezes == 3: 
65               break 
66    
67       if vezes == 3: 
68           return True 
69       else: 
70           return False
